\newtcolorbox{mytcbox}[2][]{%
	enhanced, 
	breakable,
	colback=white,
	colframe=blue!30!black,
	fontupper=\sffamily\bfseries\LARGE,
	boxrule=-1pt,
	bottom=4mm,
	top=0.35cm,
%	sharp corners=north,
%	
	fonttitle=\sffamily\bfseries, 
	attach boxed title to bottom right={yshift=10pt}, 
	boxed title size=title,
	boxed title style={%
		sharp corners, 
		rounded corners=southeast, 
		colback=tcbcolframe, 
		left=-.1cm, right=0cm,
		boxrule=0pt},
	overlay unbroken={%
		\ifthenelse{\equal{#2}{}}{
			\draw[line width=.5mm, rounded corners=\kvtcb@arc, 
			tcbcolframe] 
			(frame.south east) rectangle 
			(frame.north west);}
		{%
			\path[fill=tcbcolframe] (title.north west) 
			to[out=180, in=0] ([xshift=-1cm]title.south west) --
			(title.south west);
			\draw[line width=.5mm, rounded corners=\kvtcb@arc, 
			tcbcolframe] 
			(title.south east) rectangle 
			(frame.north west);}
	}, 
	title={#2},
	#1
}
